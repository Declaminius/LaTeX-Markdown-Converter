$$\begin{equation} \tag{3.9}
    G_{j,k}(z) 
    % &= W_{j-k}(z) - \frac{u_{1}^{j+1}(z)-u_{2}^{j+1}(z)}{u_{1}(z) - u_{2}(z)}W_{-k}(z) + u_{1}(z)u_{2}(z)\frac{u_{1}^{j}(z)-u_{2}^{j}(z)}{u_{1}(z) - u_{2}(z)}W_{-k-1}(z) \\
    = W_{j - k}(z) - G_{0,j}(z)W_{-k}(z) - z G_{1,1}(z) G_{0,j-1}(z) W_{-k-1}(z),
  \end{equation}$$
