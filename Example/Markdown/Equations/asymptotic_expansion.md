$$\begin{equation}\tag{2.9}
    \begin{split}
      \left(1+\frac{t}{n}\right)^{-n-1} 
      &= \exp\left(-(n+1)\log\left(1 + \frac{t}{n}\right)\right) \\ 
      &= \exp(-t)\left(1 + \frac{t^{2} - 2t}{2n} + \frac{3t^{4} - 20t^{3} + 24t^{2}}{24n^{2}} +  \mathcal{O}\left(\frac{1}{n^{3}}\right)\right).
    \end{split}
  \end{equation}$$
