$$\begin{equation}\tag{2.6}
    \begin{split}
      \sum_{n \geq 0} m_{n+1}(u)z^{n+1} &= zP(u)\sum_{n \geq 0} m_n(u)z^n - z\{u^{<0}\}P(u)\sum_{n \geq 0}z^n m_n(u) \\
      \iff \qquad
      M(z,u) - 1 &= zP(u)M(z,u) - z\{u^{<0}\}(P(u)M(z,u)).
    \end{split}
  \end{equation}$$
